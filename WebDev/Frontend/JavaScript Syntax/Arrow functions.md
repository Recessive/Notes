```js
const a = () => {44 + 22} // Returns nothing
const b = () => {return 44 + 22} // returns 66
const c = () => (44 + 22) // returns 66
const d = () => 44 + 22 // returns 66
const e = (param1, param2) => param1 + param2 // returns param1 + param2
```
